# Ensure results dict exists and build names from its keys
if 'results' not in globals() or not isinstance(results, dict) or len(results) == 0:
    raise RuntimeError("No 'results' dictionary found. Run model training/evaluation cells before plotting.")

names = list(results.keys())

# Colors for models (safe lookup with fallback)
COLORS_MODELS = {
    'Logistic Regression': "#B0B04C",
    'Decision Tree': '#55A868',
    'Gradient Boosting': '#8172B2'
}
default_color = "#777777"

# Safely extract metrics (use .get to avoid KeyError)
aucs = [results[n].get('roc_auc', np.nan) for n in names]
cv_means = [results[n].get('cv_mean', np.nan) for n in names]
cv_stds = [results[n].get('cv_std', np.nan) for n in names]

x = np.arange(len(names))

plt.figure(figsize=(7,5))
plt.bar(x, aucs, color=[COLORS_MODELS.get(n, default_color) for n in names])

# CV error bars: only plot when at least one valid value exists
cv_means_arr = np.array(cv_means, dtype=float)
cv_stds_arr = np.array(cv_stds, dtype=float)
if not np.all(np.isnan(cv_means_arr)):
    cv_stds_arr = np.nan_to_num(cv_stds_arr, nan=0.0)
    plt.errorbar(x, cv_means_arr, yerr=cv_stds_arr, fmt='o', color='black', capsize=5)

# Labels on bars (skip NaN values)
for i, v in enumerate(aucs):
    if np.isfinite(v):
        plt.text(i, v + 0.01, f"{v:.3f}", ha='center', fontweight='bold')

plt.axhline(0.5, linestyle='--', color='gray')

plt.xticks(x, names, rotation=15)
plt.ylabel("ROC-AUC Score")
plt.title("Model Comparison: ROC-AUC\n(Logistic vs Decision Tree vs Gradient Boosting)")

plt.ylim(0, 1)
plt.tight_layout()
plt.show()

DRIVE_FEATURES = driving_stats
PUTT_FEATURES = putting_stats

# Define the outcomes to plot (3 columns as the original layout expects)
OUTCOMES = {
    "Success": "Success",
    "avgScore": "Avg Score",
    "Points": "Points"
}

# Build a clean dataframe with only the needed columns and drop rows with NaNs for those columns
df_clean = df[DRIVE_FEATURES + PUTT_FEATURES + list(OUTCOMES.keys())].dropna()

# Create subplots based on number of outcomes
fig, axes = plt.subplots(1, len(OUTCOMES), figsize=(18, 5))
fig.suptitle('Correlation of Driving & Putting Features with Success Metrics',
             fontsize=14, fontweight='bold')

for ax, (outcome, label) in zip(axes, OUTCOMES.items()):
    corr_drive = df_clean[DRIVE_FEATURES + [outcome]].corr()[outcome].drop(outcome)
    corr_putt  = df_clean[PUTT_FEATURES  + [outcome]].corr()[outcome].drop(outcome)
    corr_all   = pd.concat([corr_drive, corr_putt]).rename('r')

    # color arrays sized to the actual number of features
    colors = ['#2196F3'] * len(DRIVE_FEATURES) + ['#FF5722'] * len(PUTT_FEATURES)
    bars = ax.barh(corr_all.index, corr_all.values, color=colors)
    ax.axvline(0, color='black', linewidth=0.8)
    ax.set_title(label, fontsize=11)
    ax.set_xlabel('Pearson r')
    ax.set_xlim(-1, 1)

plt.tight_layout(rect=[0, 0, 1, 0.95])
